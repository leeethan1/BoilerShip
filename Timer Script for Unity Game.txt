
public class CountdownTimer : MonoBehaviour
{
	float currentTime = 0f;
	float starting Time = 10f;

	void Start()
	{
		currentTime = startingTime;
	}

	void Update()
	{
		currentTime -= 1 * Time.deltaTime;
		print (currentTime);
	}
}
